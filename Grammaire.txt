# Définition des règles de la grammaire pour le langage draw++
# Format : Nom_de_Règle ::= Expression

start ::= statement+
statement ::= assignment | draw_command
assignment ::= identifier '=' expression
draw_command ::= 'draw' shape '(' parameters ')'
shape ::= 'circle' | 'rectangle' | 'line'
parameters ::= parameter (',' parameter)*
parameter ::= identifier '=' value
identifier ::= [a-zA-Z_][a-zA-Z0-9_]*
value ::= number | string
number ::= [0-9]+ ('.' [0-9]+)?
string ::= '"' [^"]* '"'



-- ex abdel gpt :
<programme> ::= <instruction> | <instruction> ";" <programme>
<instruction> ::= <mouvement> | <dessin> | <condition> | <boucle> | <assignation>
<mouvement> ::= "MOVE" <direction> <valeur>
<dessin> ::= "DRAW" <forme> <parametres>
<condition> ::= "IF" <condition_logique> "THEN" <programme> "END"
<boucle> ::= "FOR" <variable> "IN" <valeur> ".." <valeur> "DO" <programme> "END"
<assignation> ::= <variable> "=" <valeur>
<forme> ::= "CIRCLE" | "SQUARE" | "LINE"
<direction> ::= "UP" | "DOWN" | "LEFT" | "RIGHT"
<parametres> ::= "(" <valeur> "," <valeur> ")"
<valeur> ::= <nombre> | <variable>
<variable> ::= [a-zA-Z][a-zA-Z0-9_]*  // Identificateur

-- Chatgpt Myriam

<programme> ::= <instruction> | <instruction> ";" <programme>
<instruction> ::= <mouvement> 
                | <dessin> 
                | <assignation> 
                | <condition> 
                | <boucle> 
                | <bloc> 
                | <creation_curseur> 
                | <apparence_curseur>

<creation_curseur> ::= "CURSOR" <identifiant> "(" <valeur> "," <valeur> ")" [ "VISIBLE" "=" <booleen> ]
EX: CURSOR myCursor (100, 200) VISIBLE = true;

<mouvement> ::= "MOVE" <identifiant> <direction> <valeur>
EX: MOVE myCursor UP 50;

<apparence_curseur> ::= "SET" <identifiant> "COLOR" <couleur> | "THICKNESS" <valeur>
EX: SET myCursor COLOR RED;
SET myCursor THICKNESS 3;

<dessin> ::= "DRAW" <forme> "(" <identifiant> "," <parametres> ")"
<forme> ::= "LINE" | "CIRCLE" | "SQUARE" | "ARC"
<parametres> ::= <valeur> [ "," <valeur> ... ]
EX: DRAW CIRCLE(myCursor, 50);  // Cercle de rayon 50
DRAW LINE(myCursor, 100);  // Ligne de longueur 100

<rotation> ::= "ROTATE" <identifiant> <valeur>
EX: ROTATE myCursor 90;  // Tourne le curseur de 90 degrés

<animation> ::= "ANIMATE" <identifiant> "(" <parametres_animation> ")"
<parametres_animation> ::= "LOOP" | "ONCE" [ "," <vitesse> ]
EX:ANIMATE myCursor LOOP, 100ms;

<assignation> ::= <variable> "=" <expression>
<expression> ::= <valeur> | <expression> <operateur> <expression>
EX: x = 10;
y = x + 5;

<bloc> ::= "BEGIN" <programme> "END"
EX:
BEGIN
  MOVE myCursor UP 100;
  DRAW SQUARE(myCursor, 50);
END

<condition> ::= "IF" <condition_logique> "THEN" <programme> [ "ELSE" <programme> ] "END"
<condition_logique> ::= <expression> <operateur_logique> <expression>
EX:
IF x > 50 THEN
  DRAW CIRCLE(myCursor, 30);
ELSE
  DRAW SQUARE(myCursor, 50);
END

<boucle> ::= "FOR" <variable> "IN" <valeur> ".." <valeur> "DO" <programme> "END"
           | "WHILE" <condition_logique> "DO" <programme> "END"
EX:
FOR i IN 1..10 DO
  DRAW LINE(myCursor, 20);
  ROTATE myCursor 36;
END

WHILE x < 100 DO
  MOVE myCursor UP 10;
END

<valeur> ::= <nombre> | <variable>
<variable> ::= [a-zA-Z][a-zA-Z0-9_]*
<booleen> ::= "true" | "false"
<couleur> ::= "RED" | "GREEN" | "BLUE" | ...
<direction> ::= "UP" | "DOWN" | "LEFT" | "RIGHT"

<operateur> ::= "+" | "-" | "*" | "/"
<operateur_logique> ::= ">" | "<" | ">=" | "<=" | "==" | "!="
EX:
CURSOR myCursor (100, 100) VISIBLE = true;
SET myCursor COLOR RED;
MOVE myCursor UP 50;
DRAW CIRCLE(myCursor, 30);

IF x > 10 THEN
  DRAW SQUARE(myCursor, 40);
ELSE
  DRAW LINE(myCursor, 100);
END

FOR i IN 1..5 DO
  ROTATE myCursor 72;
  DRAW LINE(myCursor, 50);
END
